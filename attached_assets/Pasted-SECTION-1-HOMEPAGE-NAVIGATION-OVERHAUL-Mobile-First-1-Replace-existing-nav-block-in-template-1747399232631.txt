🧠 3. Add this JS snippet just before </body> in base.html:
<script>
  function toggleMenu() {
    const menu = document.getElementById('mobile-menu');
    menu.style.display = (menu.style.display === 'flex') ? 'none' : 'flex';
  }
</script>
