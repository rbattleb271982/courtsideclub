# Extend blog logic to match tournaments
# In routes/blog.py (or routes/main.py if blog routes live there)

from flask import render_template
from models import Tournament, BlogPost
import re

@main_bp.route('/blog/<slug>')
def blog_post(slug):
    blog = BlogPost.query.filter_by(slug=slug).first_or_404()
    tournaments = Tournament.query.all()

    matched_tournaments = []
    content_to_check = f"{blog.title} {blog.content}"

    for t in tournaments:
        if re.search(rf"\b{re.escape(t.name)}\b", content_to_check, re.IGNORECASE) or \
           re.search(rf"\b{re.escape(t.slug)}\b", content_to_check, re.IGNORECASE):
            matched_tournaments.append(t)

    blog.related_tournaments = matched_tournaments

    return render_template('blog_post.html', blog=blog)

# Tournament detail logic (routes/tournament_routes.py)

@main_bp.route('/tournaments/<slug>')
def tournament_detail(slug):
    tournament = Tournament.query.filter_by(slug=slug).first_or_404()
    blog_posts = BlogPost.query.all()

    related_blog_posts = []
    for post in blog_posts:
        if re.search(rf"\b{re.escape(tournament.name)}\b", post.content, re.IGNORECASE) or \
           re.search(rf"\b{re.escape(tournament.slug)}\b", post.content, re.IGNORECASE):
            related_blog_posts.append(post)

    return render_template('tournament_detail.html', tournament=tournament, related_blog_posts=related_blog_posts)
