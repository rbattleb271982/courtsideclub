🛠️ Step 5: Add routes for adding/removing
File: routes/user.py

@user_bp.route('/add_wishlist', methods=['POST'])
@login_required
def add_wishlist():
    tournament_id = request.form.get('tournament_id')
    existing = UserWishlistTournament.query.filter_by(user_id=current_user.id, tournament_id=tournament_id).first()
    if not existing:
        entry = UserWishlistTournament(user_id=current_user.id, tournament_id=tournament_id)
        db.session.add(entry)
        db.session.commit()
    return redirect(url_for('user.profile'))

@user_bp.route('/remove_wishlist/<int:id>', methods=['POST'])
@login_required
def remove_wishlist(id):
    entry = UserWishlistTournament.query.filter_by(id=id, user_id=current_user.id).first()
    if entry:
        db.session.delete(entry)
        db.session.commit()
    return redirect(url_for('user.profile'))
