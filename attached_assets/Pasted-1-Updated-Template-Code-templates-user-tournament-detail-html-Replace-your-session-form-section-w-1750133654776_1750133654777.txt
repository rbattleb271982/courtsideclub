class UserTournament(db.Model):
    __tablename__ = 'user_tournament'

    id = db.Column(db.Integer, primary_key=True)
    user_id = db.Column(db.Integer, db.ForeignKey('users.id'), nullable=False)
    tournament_id = db.Column(db.String(50), db.ForeignKey('tournaments.id'), nullable=False)

    session_label = db.Column(db.Text)  # e.g. "Day 1 - Day,Day 2 - Night"
    attending = db.Column(db.Boolean, default=False)
    wants_to_meet = db.Column(db.Boolean, default=True)
🧪 4. JavaScript (static/js/tournament-detail-simplified.js)
You must include this logic to handle button click → reveal form:
document.addEventListener("DOMContentLoaded", function () {
  const goingBtn = document.getElementById("btn-going");
  const maybeBtn = document.getElementById("btn-maybe");
  const notGoingBtn = document.getElementById("btn-not");
  const sessionForm = document.getElementById("session-selection-ui");
  const attendanceInput = document.getElementById("attendance-status");

  if (goingBtn && sessionForm) {
    goingBtn.addEventListener("click", function (e) {
      e.preventDefault();
      goingBtn.classList.add("btn-success");
      maybeBtn.classList.remove("btn-success");
      notGoingBtn.classList.remove("btn-success");

      attendanceInput.value = "attending";
      sessionForm.style.display = "block";
    });
  }

  if (notGoingBtn && sessionForm) {
    notGoingBtn.addEventListener("click", function (e) {
      e.preventDefault();
      goingBtn.classList.remove("btn-success");
      maybeBtn.classList.remove("btn-success");
      notGoingBtn.classList.add("btn-success");

      attendanceInput.value = "not_attending";
      sessionForm.style.display = "none";
    });
  }

  if (maybeBtn && sessionForm) {
    maybeBtn.addEventListener("click", function (e) {
      e.preventDefault();
      maybeBtn.classList.add("btn-success");
      goingBtn.classList.remove("btn-success");
      notGoingBtn.classList.remove("btn-success");

      attendanceInput.value = "maybe";
      sessionForm.style.display = "block";
    });
  }
});
