. Create a Shared Hero Component
Create a new file:
/components/shared/HeroSection.tsx
Paste in this component:
"use client"
import Image from "next/image"

type HeroSectionProps = {
  topText: string
  highlightText: string
  subtitle?: string
  backgroundImage?: string
  primaryCTA?: React.ReactNode
  secondaryCTA?: React.ReactNode
}

export default function HeroSection({
  topText,
  highlightText,
  subtitle,
  backgroundImage = "/backgrounds/stadium-sunset.jpg",
  primaryCTA,
  secondaryCTA,
}: HeroSectionProps) {
  return (
    <section className="relative py-32 flex items-center justify-center overflow-hidden">
      <div className="absolute inset-0 z-0">
        <Image
          src={backgroundImage}
          alt="Hero Background"
          layout="fill"
          objectFit="cover"
          className="opacity-50"
        />
        <div className="absolute inset-0 bg-gradient-to-b from-black/70 via-black/60 to-black/80" />
      </div>

      <div className="relative z-10 text-center px-6 max-w-4xl mx-auto">
        <h1 className="text-5xl md:text-6xl font-light text-white mb-6">
          {topText}
          <span className="block font-black text-transparent bg-gradient-to-r from-lime-400 via-blue-400 to-amber-400 bg-clip-text">
            {highlightText}
          </span>
        </h1>
        {subtitle && (
          <p className="text-lg md:text-xl text-slate-300 font-medium leading-relaxed max-w-2xl mx-auto">
            {subtitle}
          </p>
        )}
        {(primaryCTA || secondaryCTA) && (
          <div className="mt-10 flex justify-center gap-4">
            {primaryCTA}
            {secondaryCTA}
          </div>
        )}
      </div>
    </section>
  )
}
