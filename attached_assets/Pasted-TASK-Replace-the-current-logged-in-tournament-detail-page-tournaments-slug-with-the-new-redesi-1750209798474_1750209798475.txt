Create or update to include:
function setAttendance(type) {
  document.getElementById('attendance-status').value = type;

  document.querySelectorAll('.attendance-toggle').forEach(btn => {
    btn.classList.remove('selected');
  });
  if (type === 'attending') {
    document.querySelectorAll('.attendance-toggle')[0].classList.add('selected');
  } else if (type === 'maybe') {
    document.querySelectorAll('.attendance-toggle')[1].classList.add('selected');
  } else {
    document.querySelectorAll('.attendance-toggle')[2].classList.add('selected');
  }

  const sessionSection = document.getElementById('session-section');
  if (sessionSection) {
    sessionSection.style.display = type === 'not_attending' ? 'none' : 'block';
  }
}
