2. Create or Update mark_attendance() Route
In routes/user.py, add or confirm a route like:
@user_bp.route('/tournaments/attend/<int:tournament_id>', methods=['POST'])
@login_required
def mark_attendance(tournament_id):
    status = request.form.get('status')
    tournament = Tournament.query.get_or_404(tournament_id)

    user_tourney = UserTournament.query.filter_by(user_id=current_user.id, tournament_id=tournament_id).first()

    if not user_tourney:
        user_tourney = UserTournament(user_id=current_user.id, tournament_id=tournament_id)

    user_tourney.attending = (status == "attending")
    user_tourney.session_label = ""  # reset for now until session is selected

    db.session.add(user_tourney)
    db.session.commit()

    return redirect(url_for('tournaments.detail', slug=tournament.slug))
