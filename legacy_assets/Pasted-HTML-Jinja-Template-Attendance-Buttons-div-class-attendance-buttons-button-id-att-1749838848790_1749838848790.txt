✅ JavaScript
<script>
  const attendingBtn = document.getElementById('attending-btn');
  const maybeBtn = document.getElementById('maybe-btn');
  const sessionUI = document.getElementById('session-selection');
  const statusInput = document.getElementById('attendance-status');

  function showSessionUI(status) {
    sessionUI.style.display = 'block';
    statusInput.value = status;
    attendingBtn.classList.remove('selected');
    maybeBtn.classList.remove('selected');

    if (status === 'attending') attendingBtn.classList.add('selected');
    if (status === 'maybe') maybeBtn.classList.add('selected');
  }

  attendingBtn?.addEventListener('click', () => showSessionUI('attending'));
  maybeBtn?.addEventListener('click', () => showSessionUI('maybe'));
</script>
